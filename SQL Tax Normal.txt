SELECT
  t.tahun,
  SUM(t.pendapatan) AS total_pendapatan,
  SUM(t.beban_operasional + t.penyusutan) AS total_beban,
  SUM(t.pendapatan - t.beban_operasional - t.penyusutan) AS laba_kena_pajak,
  k.tax_rate,
  SUM(t.pendapatan - t.beban_operasional - t.penyusutan) * k.tax_rate AS pph_badan
FROM
  `modified-talon-457803-m2.PPh_Badan.Transaksi Keuangan` t
JOIN
  `modified-talon-457803-m2.PPh_Badan.Kebijakan Fiskal` k
ON
  t.tahun = k.tahun
WHERE
  t.skenario = 'normal'
GROUP BY
  t.tahun, k.tax_rate
ORDER BY
  t.tahun;
